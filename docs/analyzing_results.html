def class_label_mapper(label, map_dict, map_dict_field):
    if label in map_dict:

def count_ngrams(instance_set, text_model):
    ngram_counts = defaultdict(int)
    
    for i in instance_set:
        feats = text_model.text_converter.feat_gen.unigram(text_model.text_converter.text_prep.preprocess(i.text))
        ngram_freq = {text_model.text_converter.feat_gen.fidx2ngram[k]:v for k,v in feats.iteritems()}
        readable = {convert_ngram(k):v for k,v in ngram_freq.iteritems()}
        for k,v in readable.iteritems():
            ngram_counts[k] += v
    
    return ngram_counts
